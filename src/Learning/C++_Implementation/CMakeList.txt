cmake_minimum_required(VERSION 3.10)
project(SNN_CUDA LANGUAGES CXX CUDA)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Set CUDA standard
set(CMAKE_CUDA_STANDARD 17)
set(CMAKE_CUDA_STANDARD_REQUIRED ON)

# Find CUDA
find_package(CUDA REQUIRED)

# Include directories
include_directories(${CUDA_INCLUDE_DIRS})

# Add executable
add_executable(snn_cuda snn_cuda.cu)

# Link CUDA libraries
target_link_libraries(snn_cuda 
    ${CUDA_CUDART_LIBRARY}
    ${CUDA_CUBLAS_LIBRARIES}
    ${CUDA_CURAND_LIBRARY}
)

# CUDA compilation flags
set_target_properties(snn_cuda PROPERTIES
    CUDA_SEPARABLE_COMPILATION ON
    CUDA_ARCHITECTURES 75 80 86 89  # Adjust to your GPU
)

# Optimization flags
target_compile_options(snn_cuda PRIVATE
    $<$<COMPILE_LANGUAGE:CUDA>:
        --use_fast_math
        -Xcompiler -O3
    >
    $<$<COMPILE_LANGUAGE:CXX>:
        -O3 -march=native
    >
)
