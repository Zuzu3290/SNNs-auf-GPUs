# Compile with Vivado
vivado -mode batch -source synthesize.tcl

# Or with iverilog (for simulation)
iverilog -o snn_core lif_neuron_core.v
vvp snn_core

# For synthesis to Xilinx FPGA
vivado -mode batch \
  -source <(echo "
    create_project snn_fpga
    add_files lif_neuron_core.v
    set_property top_module snn_core [current_fileset]
    launch_synthesis
    launch_implementation
  ")
